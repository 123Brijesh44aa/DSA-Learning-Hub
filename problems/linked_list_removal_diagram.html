‚òï Java Implementation
/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    // Remove all nodes with target value from linked list
    // Time: O(n), Space: O(1)
    public ListNode removeElements(ListNode head, int val) {
        // Phase 1: Handle leading nodes to remove
        while (head != null && head.val == val) {
            head = head.next;
        }
        
        // Phase 2: Handle remaining nodes with two pointers
        ListNode current = head;
        ListNode prev = null;
        
        while (current != null) {
            if (current.val == val) {
                prev.next = current.next; // Skip current node
            } else {
                prev = current; // Move prev forward
            }
            current = current.next;
        }
        
        return head;
    }
}

 *     ListNode *next;

    // Remove all nodes with target value from linked list
    // Time: O(n), Space: O(1)
    ListNode* removeElements(ListNode* head, int val) {
        // Phase 1: Handle leading nodes to remove
        while (head != nullptr && head->val == val) {
            ListNode* temp = head;
            head = head->next;
            delete temp; // Free memory in C++
        }

        ListNode* current = head;

                prev->next = current->next;
                delete current; // Free memory
                current = prev->next;
            } else {
                prev = current;
                current = current->next;
            }
        }
        
        return head;
    }
};

func removeElementsRecursive(head *ListNode, val int) *ListNode {

// Definition for singly-linked list.
// #[derive(PartialEq, Eq, Clone, Debug)]

//   pub val: i32,
//   pub next: Option>

//

//   #[inline]
//   fn new(val: i32) -> Self {
//     ListNode {
//       next: None,

//     }

impl Solution {
    // Remove all nodes with target value from linked list
    // Time: O(n), Space: O(1)
    pub fn remove_elements(head: Option<Box<ListNode>>, val: i32) -> Option<Box<ListNode>> {
        // Handle leading nodes to remove
        let mut head = head;
        while let Some(node) = head {
            if node.val == val {
                head = node.next;
            } else {
                break;
            }
        }
        
        // Handle remaining nodes
        if let Some(ref mut node) = head {
            node.next = Self::remove_elements(node.next.take(), val);
        }
        
        head
    }
    
    // Alternative iterative approach
    pub fn remove_elements_iterative(head: Option<Box<ListNode>>, val: i32) -> Option<Box<ListNode>> {
        let mut dummy = Box::new(ListNode::new(0));
        dummy.next = head;
        let mut prev = &mut dummy;
        
        while let Some(ref mut current) = prev.next {
            if current.val == val {
                prev.next = current.next.take();
            } else {
                prev = prev.next.as_mut().unwrap();
            }
        }
        
        dummy.next
    }
}
